function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); var data = JSON.parse(e.postData.contents); var custom = data.customData || {}; var order = data.order || {}; var lineItemsArr = order.line_items || []; var customer = order.customer || {}; // Build line items string var itemsList = lineItemsArr.map(function(item) { return item.title + " x" + item.quantity; }).join(", "); // Get order ID from line_items meta var orderId = ""; if (lineItemsArr.length > 0 && lineItemsArr[0].meta) { orderId = lineItemsArr[0].meta.order_id || ""; } // Customer info var customerName = custom.customer_name || data.full_name || customer.name || ""; var customerPhone = custom.customer_phone || data["Billing Address - Phone Number"] || ""; // Pickup info var pickupTime = custom.pickup_time || ""; var time = custom.time || (data["Pick a Time"] ? data["Pick a Time"][0] : "") || ""; var date = custom.date || data["Choose your Day"] || ""; // Notes and mods var notes = custom.notes || data["Special Requests"] || ""; var mods = custom.mods || ""; var newRow = sheet.getLastRow() + 1; // Write timestamp with full date AND time var now = new Date(); sheet.getRange(newRow, 1).setValue(now); sheet.getRange(newRow, 1).setNumberFormat('M/d/yyyy H:mm:ss'); sheet.getRange(newRow, 2).setValue(orderId); sheet.getRange(newRow, 3).setValue(customerName); sheet.getRange(newRow, 4).setValue(customerPhone); sheet.getRange(newRow, 5).setValue(pickupTime); sheet.getRange(newRow, 6).setValue(itemsList); sheet.getRange(newRow, 7).setValue(notes); sheet.getRange(newRow, 8).setValue(mods); sheet.getRange(newRow, 9).setValue(time); sheet.getRange(newRow, 10).setValue(date); sheet.getRange(newRow, 11).setValue("🔴 NEW"); // Formatting sheet.getRange(newRow, 1, 1, 11) .setFontSize(13) .setFontWeight("bold") .setVerticalAlignment("middle"); sheet.getRange(newRow, 1, 1, 11).setBackground("#FFF176"); sheet.getRange(newRow, 6).setWrap(true); sheet.getRange(newRow, 7).setWrap(true); sheet.getRange(newRow, 8).setWrap(true); sheet.setRowHeight(newRow, 80); return ContentService .createTextOutput(JSON.stringify({status: "success", received: data})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status: "error", message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput("KDS Webhook is live.") .setMimeType(ContentService.MimeType.TEXT); }
